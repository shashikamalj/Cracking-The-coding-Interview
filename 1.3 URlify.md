string urlify(string str) {
    for (int i = 0; i < str.size(); ) {
        if (isblank(str[i])) {
            int final = str.size() - 1;
            str.resize(str.size() + 2);
            int newfinal = str.size() - 1;
            for (int j = final; j != i; j--) {
                str[newfinal--] = str[j];
            }

            string temp = "%02";
            str.replace(i, 3, temp);
            //str.replace(i, temp);
            i += 3;
        }
        else
          i++;
    }

    return str;
}
